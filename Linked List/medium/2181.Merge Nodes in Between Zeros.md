## Solution 1: In-Place Modification -->  TC: O(n) / SC: O(1)
class Solution:
    def mergeNodes(self, head: Optional[ListNode]) -> Optional[ListNode]:
        p1=head ; p2=head.next ; sm=0
        while p2:
            if p2.val:
                sm+=p2.val
            else:
                p1=p1.next
                p1.val=sm
                sm=0
            p2=p2.next
        p1.next=None
        return head.next

                                                                                                                           

                                                                                                                           
## Solution 2:  TC: O(n) / SC: O(k)   (where k: len(dummy))
class Solution:
    def mergeNodes(self, head: Optional[ListNode]) -> Optional[ListNode]:
        ans=dummy=ListNode()
        curr=head.next
        while curr:
            sm=0
            while curr and curr.val:
                sm+=curr.val
                curr=curr.next
            ans.next=ListNode(sm,None)
            ans=ans.next
            curr=curr.next
        return dummy.next
