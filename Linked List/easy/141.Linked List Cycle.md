class Solution:
    def hasCycle(self, head: Optional[ListNode]) -> bool:
        Seen=set()
        while head:
            if head in Seen: return True
            Seen.add(head)
            head=head.next
                                                                                                                           

                                                                                                                           
## Solution :  TC: O(n) / SC: O(1)
class Solution:                                                                                                                     
    def hasCycle(self, head: Optional[ListNode]) -> bool:
        slow=fast=head
        while slow and fast and fast.next:
            slow=slow.next
            fast=fast.next.next
            if slow==fast: return True
