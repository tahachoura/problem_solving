## Solution 1:  TC: O(n) / SC: O(1)    
class Solution:
    def getDecimalValue(self, head: ListNode) -> int:
        ans=0
        while head: 
            ans=(ans<<1) + head.val
            head=head.next
        return ans
                                                                                                                           

                                                                                                                           
## Solution 2:  TC: O(n) / SC: O(1)
class Solution:
    def getDecimalValue(self, head: ListNode) -> int:
        ans=0
        while head: 
            ans=ans*2+head.val
            head=head.next
        return ans
